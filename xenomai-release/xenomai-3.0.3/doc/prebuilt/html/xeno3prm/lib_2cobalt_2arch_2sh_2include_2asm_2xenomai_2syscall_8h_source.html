94 #define XENOMAI_DO_SYSCALL(nr, op, args...) \
95  ({ \
96  unsigned long int __ret; \
97  register long int r3 asm ("%r3") = __xn_syscode(op); \
98  SUBSTITUTE_ARGS_##nr(args); \
99  \
100  asm volatile (SYSCALL_INST_STR##nr SYSCALL_INST_PAD \
101  : "=z" (__ret) \
102  : "r" (r3) ASMFMT_##nr \
103  : "memory", "t"); \
104  \
105  (int) __ret; \
106  })
